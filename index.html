    <!-- Quill.js JS -->

    <script src="https://cdn.quilljs.com/1.3.6/quill.js"></script>


    <script>

        // --- 모바일 메뉴 토글 ---

        const mobileMenuButton = document.getElementById('mobile-menu-button');

        const mobileMenu = document.getElementById('mobile-menu');

        mobileMenuButton.addEventListener('click', () => {

            mobileMenu.classList.toggle('hidden');

        });


        // --- 경력 기술서 업데이트 기능 ---

        const resumeInput = document.getElementById('resume-input');

        const updateResumeBtn = document.getElementById('update-resume-btn');

        const resumeOutput = document.getElementById('resume-output');


        updateResumeBtn.addEventListener('click', () => {

            const resumeText = resumeInput.value.trim();

            if (!resumeText) {

                resumeOutput.innerHTML = '<p class="text-gray-500 dark:text-gray-400">경력 기술서 내용을 입력하고 "업데이트" 버튼을 눌러주세요.</p>';

                return;

            }

            const preElement = document.createElement('pre');

            preElement.textContent = resumeText;

            preElement.className = 'whitespace-pre-wrap font-mono text-left';

            resumeOutput.innerHTML = '';

            resumeOutput.appendChild(preElement);

        });


        // --- 게시판 기능 ---

        const newPostBtn = document.getElementById('new-post-btn');

        const editorModal = document.getElementById('editor-modal');

        const cancelPostBtn = document.getElementById('cancel-post-btn');

        const savePostBtn = document.getElementById('save-post-btn');

        const postTitleInput = document.getElementById('post-title');

        const postsContainer = document.getElementById('posts-container');

        const noPostsMsg = document.getElementById('no-posts-msg');

        

        let quill;

        let posts = []; // 게시글 데이터를 저장할 배열


        // Quill 편집기 초기화 함수

        function initializeQuill() {

            if (!quill) {

                quill = new Quill('#editor', {

                    modules: {

                        toolbar: [

                            [{ header: [1, 2, 3, false] }],

                            ['bold', 'italic', 'underline', 'strike'],

                            ['link', 'image', 'blockquote', 'code-block'],

                            [{ list: 'ordered' }, { list: 'bullet' }]

                        ]

                    },

                    placeholder: '내용을 작성해주세요...',

                    theme: 'snow'

                });

            }

        }


        // 글쓰기 모달 열기

        newPostBtn.addEventListener('click', () => {

            editorModal.classList.remove('hidden');

            initializeQuill();

        });


        // 글쓰기 모달 닫기

        function closeModal() {

            postTitleInput.value = '';

            quill.root.innerHTML = '';

            editorModal.classList.add('hidden');

        }


        cancelPostBtn.addEventListener('click', closeModal);


        // 게시글 저장

        savePostBtn.addEventListener('click', () => {

            const title = postTitleInput.value.trim();

            const content = quill.root.innerHTML;

            

            if (!title) {

                alert('제목을 입력해주세요.');

                return;

            }


            const now = new Date();

            const post = {

                id: Date.now(),

                title: title,

                content: content,

                date: now.toLocaleDateString('ko-KR')

            };


            posts.unshift(post); // 새 글을 배열 맨 앞에 추가

            renderPosts();

            closeModal();

        });


        // 게시글 목록 렌더링

        function renderPosts() {

            postsContainer.innerHTML = ''; // 기존 목록 초기화


            if (posts.length === 0) {

                postsContainer.appendChild(noPostsMsg);

                noPostsMsg.classList.remove('hidden');

            } else {

                noPostsMsg.classList.add('hidden');

                posts.forEach(post => {

                    const postElement = document.createElement('div');

                    postElement.className = 'bg-white dark:bg-gray-800 p-6 rounded-lg shadow-md';

                    postElement.innerHTML = `

                        <div class="flex justify-between items-center mb-2">

                            <h3 class="text-xl font-bold text-gray-900 dark:text-white">${post.title}</h3>

                            <p class="text-sm text-gray-500 dark:text-gray-400">${post.date}</p>

                        </div>

                        <div class="prose dark:prose-invert max-w-none ql-editor p-0">${post.content}</div>

                    `;

                    postsContainer.appendChild(postElement);

                });

            }

        }

    </script>
